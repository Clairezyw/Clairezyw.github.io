webpack简单搭建vue架构

Veröffentlicht am 2018-05-30 | in webpack . vue | 浏览 次

1.安装

mkdir vue-demo //创建vue-demo包
cd vue-demo  
npm init -y   
npm install webpack webpack-cli --save-dev //下载webpack webpack-cli

创建如下目录

 vue-demo
 |-package.json
 |-node_modules
+|-src 
   |-router //路由相关配置
     |-index.js
   |-main.js
   |-App.vue
+|-dist
   |-index.html
+|-webpack.config.js

2.下载相关依赖

npm install babel-loader css-loader style-loader file-loader vue vue-loader vue-router webpack vue-template-compiler --save-dev

3.相关配置

webpack.config.js
 const path = require('path');
 const {VueLoaderPlugin} = require('vue-loader');
 const webpack = require('webpack');

 module.exports = {
    entry:'./src/main.js',
    output: {
        filename: 'bundle.js',//build之后生成的js文件，名字自己起
        path: path.resolve(__dirname, 'dist')
    },
    devServer: {
        contentBase: './dist',
        hot true//实时刷新
    },
    plugins: [
        new VueLoaderPlugin(),//使用vue-loader插件
        new webpack.HotModuleReplacementPlugin()//使用模块热替换
    ],
    module: {
        rules: [
        {
            test: /\.vue$/,
            use: 'vue-loader'//解析.vue文件
        },{
            test: /\.css$/,
            use: ['css-loader', 'style-loader']//解析css
        },{
            test: /\.js$/,
            use: 'babel-loader',//使用babel将es6解析为浏览器认识的es5
            exclude: /node_modules/
        },{
            test: /\.(png|svg|jpg|gif)$/,
            use: 'file-loader'//解析图片
        }
        ]
    },
    resolve: {
        alias: {
            'vue$': 'vue/dist/vue.esm.js'
        }
    },//参照官网
}
dist/html
 <!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <title>Document</title>
</head>
<body>
    <div id="app"></div>
    <script src="bundle.js"></script>
</body>
</html>
src/router/index.js 配置路由
import Vue from 'vue';
import Router from 'vue-router';
import App from '../App.vue';


Vue.use(Router);
export default new Router({
    routes: [
        {
            path: '/',
            name: 'index',
            component: App
        }
    ]
})
src/main.js 全局配置
import Vue from 'vue';
import App from './App.vue';
import VueRouter from 'vue-router';
import router from './router';

Vue.config.debug = true;

new Vue({
    el: "#app",
    router: router,
    template: '<App/>',
    render: h => h(App)
})
src/App.vue
<template>
    <div>Hello {{msg}}</div>
</template>
<script>
export default {
    data () {
        return {
            msg: 'Vue'
        }
    }
}
</script>
package.json 配置dev、build
  "scripts": {
    "test": "echo \"Error: no test specified\" && exit 1",
    "dev": "webpack-dev-server --mode=development",
    "build": "webpack --mode=production"
  },

4.run

npm run build
npm run dev

打开localhost:8080网址看如果浏览器出现“Hello Vue”说明配置成功。
