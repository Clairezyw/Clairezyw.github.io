set与map

Veröffentlicht am 2017-12-19 | in ECMAScript . set&&map | 浏览 次
set

es6提供了新的数据结构set，该结构有点类似于数组，但其值是不可重复的。
下面来看一下常用的方法：
1、add（） 添加一个元素
2、delete（）删除一个元素
3、has（） 判断是否包含某个元素
4、clear（）清空所有的元素

let arr = ['name:claire','age:18','age:18'];
let st  = new Set(arr);
//age：18　重复，只保留一个age
console.log(Array.from(st));//name:claire age:18
//has方法
console.log(st.has('name:claire'));//true
　//delete
st.delete("age:18");
console.log(st);//name:claire
st.add(22);
console.log(st)//name:claire age:18 22
//clear()
st.clear();//st为空

除此之外，set还有一些常用的遍历方法

//keys
let items = st.keys();
for(let i of items){
    console.log(i);
}//name:claire age:18

let values = st.values();
for(let i of values){
    console.log(i);
}//name:claire age:18

//key和value的值一样所以结果显示两遍
 let entry = st.entries();
for(let i of entry){
    console.log(i);
}// ['name:claire', 'name:claire' ] [ 'age:18', 'age:18' ]

小结：set的特点是不允许有重复的值，并且还有增删、判断是否存在、清空的方法。可通过keys、values和entries进行遍历

map

与set类似，但map存储的是键值对，其中键可以为任意类型

 //初始化map
 const map = new Map([['name','claire']]);
 map.has(name)//true
 map.get(name)//claire

map的方法：
1、set（key，value） 添加一个元素
2、delete（key）删除一个元素
3、has（key） 判断是否包含某个元素
4、clear（）清空所有的元素
5、get（key）获取键值

遍历map的方法：
1、Object.keys()
2、values
3、entries
4、forEach
用法与map类似
下面来介绍一下map之间的转换

//map转换为数组
const a = new Map();
map.set(1, 'num1')
    .set(2, 'num2');

for (let i of [...map]) {
    console.log(i);
}//[ 1, 'num1' ] [ 2, 'num2' ]
//map转换为对象，前提是map的所有键都是字符串
function strToObj(strMap) {
    let obj = Object.create(null);
    for (let [k, v] of strMap) {
        obj[k] = v;
    }
    console.log(obj);

}

const strMap = new Map().set('yes', 'true')
    .set('no', 'false');
strToObj(strMap);//{ yes: 'true', no: 'false' }
//对象转换为数组
 function objToStr(obj) {
     let strMap = new Map();
     for(let k of Object.keys(obj)){
         strMap.set(k,obj[k]);
     }
